  { pose: Flexi.confident, message: "I'm ready to learn!" },
  { pose: Flexi.teacher, message: "Let me explain this concept..." },
  { pose: Flexi.excited, message: "This is amazing!" },
  { pose: Flexi.thumbs_up, message: "Great job!" },
  { pose: Flexi.confused, message: "Need help with this?" }
